Blok program merupakan kumpulan dari statement dan ekspresi yang dibungkus menjadi satu kesatuan. Blok program selalu dibuka dengan tanda kurung kurawal { dan ditutup dengan tanda kurung kurawal }. Contoh: 

package id.ac.ub.filkom.codemaniac.materijava;
    public class BlokProgram {
        // blok program main
        public static void main(String args[]) {
            System.out.println("Hello World");
            System.out.println("Code Maniac");
    
            // blok program if
            if (true) {
                System.out.println("True");
            }
    
            // blok program for
            for (int i = 0; i < 10; i++) {
                System.out.println("Perulangan ke" + i);
            }
        }
    }
